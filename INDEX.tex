The INDEX pdf is compiled with \texttt{latexmk}
\begin{verbatim}
sudo apt install latexmk texlive-latex-extra texlive-bibtex-extra biber
latexmk --outdir=./build --pdf INDEX.tex && cp build/INDEX.pdf .
\end{verbatim}
